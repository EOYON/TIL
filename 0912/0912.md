```python
def bfs(z, y, x):
    global T
    Q = []
    Q.append([z, y, x])
    while Q:
        z, y, x = Q.pop()
        for k in range(6):
            nx = x + dx[k]
            ny = y + dy[k]
            nz = z + dz[k]
            if 0 <= nx < M and 0 <= ny < N and 0 <= nz < H:
                if arr[nz][ny][nx] == 0:
                    T.add((nz, ny, nx))


M, N, H = map(int, input().split())
arr = [[list(map(int, input().rstrip().split())) for _ in range(N)] for _ in range(H)]
dx = [0, 1, 0, -1, 0, 0]
dy = [-1, 0, 1, 0, 0, 0]
dz = [0, 0, 0, 0, 1, -1]
RT = []
URT = 0
for x in range(M):
    for y in range(N):
        for z in range(H):
            if arr[z][y][x] == 1:
                RT.append([z, y, x])
            elif arr[z][y][x] == 0:
                URT += 1

result = 0
while True:
    T = set()
    while RT:
        z, y, x = RT.pop()
        bfs(z, y, x)
    if T:
        result += 1
        while T:
            z, y, x = T.pop()
            arr[z][y][x] = 1
            RT.append([z, y, x])
            URT -= 1
    else:
        if URT:
            result = -1
        break
print(result)
```

## 3. B2589 보물섬

```python
from collections import deque
import sys
input = sys.stdin.readline
sys.setrecursionlimit(10**5)


def bfs(y, x, l):
    global result
    visited[y][x] = 1
    Q = deque()
    Q.append([y, x, l])
    while Q:
        y, x, l = Q.popleft()
        for k in range(4):
            nx = x + dx[k]
            ny = y + dy[k]
            if 0 <= nx < N and 0 <= ny < M:
                if arr[ny][nx] == 'L' and visited[ny][nx] == 0:
                    Q.append([ny, nx, l+1])
                    visited[ny][nx] = 1
                    result = l+1 if l+1 > result else result
M, N = map(int, input().split())
arr = [list(input().rstrip()) for _ in range(M)]
dx = [0, 1, 0, -1]
dy = [-1, 0, 1, 0]
result = 0
for y in range(M):
    for x in range(N):
        if arr[y][x] == 'L':
            visited = [[0] * N for _ in range(M)]
            bfs(y, x, 0)
print(result)
```

## 4. B14889 스타트와 링크

```python
from itertools import combinations
import sys
input = sys.stdin.readline

N = int(input())
arr = [list(map(int, input().rstrip().split())) for _ in range(N)]
n = N // 2
C = list(combinations(range(1, N), n-1))
result = 100 * n
for A in C:
    S = 0
    A = list(A)
    A.append(0)
    B = list(filter(lambda x:x not in A, range(N)))
    for i in range(n-1):
        for j in range(i+1, n):
            S += arr[A[i]][A[j]]
            S += arr[A[j]][A[i]]
            S -= arr[B[i]][B[j]]
            S -= arr[B[j]][B[i]]
    result = abs(S) if abs(S) < result else result
print(result)
```

## 5. B19535 ㄷㄷㄷㅈ

```python
import sys
sys.setrecursionlimit(10**5)
input = sys.stdin.readline

N = int(input())
V = [0] * (N+1)
M = []
visited = [0] * (N+1)
D = G = 0

for _ in range(N-1):
    a, b = map(int, input().split())
    M.append([a, b])
    V[a] += 1
    V[b] += 1

for n in range(1, N+1):
    lv = V[n]
    if lv >= 3:
        G += lv*(lv-1)*(lv-2)//6

for m in M:
    a, b = m
    if V[a] >= 2 and V[b] >= 2:
        D += (V[a]-1) * (V[b]-1)

if D > 3*G:
    print('D')
elif D < 3*G:
    print('G')
else:
    print('DUDUDUNGA')
```
