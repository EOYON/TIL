## 1. B2146 다리 만들기

```python
from collections import deque
import sys
sys.setrecursionlimit(10**6)
input = sys.stdin.readline

def bfs(i, j, L):
    global result
    Q = deque()
    S = deque()
    Q.append((i, j))
    while Q:
        y, x = Q.popleft()
        for k in range(4):
            ny, nx = y+dy[k], x+dx[k]
            if 0 <= ny < N and 0 <= nx < N:
                if arr[ny][nx] == 1 and visited[ny][nx] == 0:
                    visited[ny][nx] = L
                    Q.append((ny, nx))
                elif arr[ny][nx] == 0 and visited[ny][nx] != L:
                    visited[ny][nx] = L
                    S.append((ny, nx, 1))
    while S:
        y, x, n = S.popleft()
        for k in range(4):
            ny, nx = y+dy[k], x+dx[k]
            if 0 <= ny < N and 0 <= nx < N:
                if arr[ny][nx] == 0 and visited[ny][nx] != L:
                    visited[ny][nx] = L
                    S.append((ny, nx, n+1))
                elif arr[ny][nx] == 1 and visited[ny][nx] !=  L:
                    result = n if n < result else result
                    break

N = int(input())
arr = [list(map(int, input().split())) for _ in range(N)]
visited = [[0] * N for _ in range(N)]
dy = [-1, 0, 1, 0]
dx = [0, 1, 0, -1]
result = 2 * N
L = 1
for i in range(N):
    for j in range(N):
        if arr[i][j] == 1 and visited[i][j] == 0:
            visited[i][j] = L
            bfs(i, j, L)
            L += 1
print(result)
```
