## 1. B25512 트리를 간단하게 색칠하는 최소 비용

```python
import sys
sys.setrecursionlimit(10**6)

def wb(n):    # bfs 형식으로 계산
    global W
    global B
    Q = []
    Q.append([n, 0])   # 노드와 레이어(해당 노드의 깊이)를 같이 기록
    while Q:
        n, l = Q.pop()
        W += cost[n][l%2]
        B += cost[n][(l+1)%2]    # 레이어에 따라 다른 경우의 수 계산
        while ch[n]:
            m = ch[n].pop()
            Q.append([m, l+1])
    return W if W < B else B    # 마지막에 비교하여 리턴

n = int(input())
ch = [[] for _ in range(n)]
cost = [[] for _ in range(n)]
for _ in range(n-1):
    p, c = map(int, input().split())
    ch[p].append(c)
for i in range(n):
    w, b = map(int, input().split())
    cost[i].append(w)
    cost[i].append(b)
W = B = 0     # 정점 0이 검은색인 경우, 흰색인 경우 2가지로 나누어짐
print(wb(0))(b)
W = B = 0
print(wb(0))
```

- 루트에 따라 2가지 경우의 수가 나옴
  
  - 트리의 레이어를 기록하며 2가지 경우를 계산, 마지막에 비교 및 출력

## 2. 이진탐색

```python
def inorder(n):
    global cnt
    global result
    global root
    if 0 < n <= N:
        inorder(2*n)
        cnt += 1
        if n == N//2:
            result = cnt
        if n == 1:
            root = cnt
        inorder(2*n+1)

T = int(input())
for tc in range(1, T+1):
    N = int(input())
    root = 0
    cnt = result = 0
    inorder(1)
    print(f'#{tc} {root} {result}')
```

## 3. 노드의 합

```python
T = int(input())
for tc in range(1, T+1):
    N, M, L = map(int, input().split())
    val = [0] * (N+1)
    for _ in range(M):
        n, v = map(int, input().split())
        while n != 1:
            val[n] += v
            n //= 2
    print(f'#{tc} {val[L]}')
```

## 4. 이진 힙

```python
T = int(input())
for tc in range(1, T+1):
    N = int(input())
    arr = list(map(int, input().split()))
    nodes = [0] * (N+1)
    for i in range(N):
        n = i+1
        p = n//2
        nodes[n] = arr[i]
        while nodes[p] > nodes[n]:
            nodes[n], nodes[p] = nodes[p], nodes[n]
            p, n = p//2, p
    result = -nodes[N]
    while nodes[N]:
        result += nodes[N]
        N //= 2
    print(f'#{tc} {result}')
```
